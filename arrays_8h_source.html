373  /**
374  * @brief Computes the Frobenius norm of the tensor
375  *
376  * fro_norm = \f$\sqrt{ \sum_{i=0}^{I_tot}\sum_{j=0}^{J_tot}\sum_{k=0}^{K_tot} |t[k][j][i]|^2 }\f$
377  */
378  double frobenius() {
379  T norm_val = 0;
380  for (int i1 = 0; i1 < n_layers; i1++) {
381  for (int i2 = 0; i2 < n_cols; i2++) {
382  for (int i3 = 0; i3 < n_rows; i3++) { norm_val += abs(tensor[i1][i2][i3]) * abs(tensor[i1][i2][i3]); }
383  }
384  }
385  return sqrt(norm_val);
386  }
