58 /**
59 * @brief Allocates x, y, and z as (K_total+1) * (J_total+1) * (I_total+1)
60 * arrays
61 *
62 * @param I_total,J_total,K_total Dimensions of the tensor size to set
63 */
64 void allocate(int I_total, int J_total, int K_total) {
65 x = (T ***) malloc(K_total * sizeof(T **));
66 y = (T ***) malloc(K_total * sizeof(T **));
67 z = (T ***) malloc(K_total * sizeof(T **));
68 for (int k = 0; k < K_total; k++) {
69 x[k] = (T **) malloc(J_total * sizeof(T *));
70 y[k] = (T **) malloc(J_total * sizeof(T *));
71 z[k] = (T **) malloc(J_total * sizeof(T *));
72 for (int j = 0; j < J_total; j++) {
73 x[k][j] = (T *) malloc(I_total * sizeof(T));
74 y[k][j] = (T *) malloc(I_total * sizeof(T));
75 z[k][j] = (T *) malloc(I_total * sizeof(T));
76 }
77 }
78 }

232template<typename T>
233class Vector {
234protected:
235 int n = 0; // Number of elements
236 T *vector = nullptr;// Internal array
237
238public:
239 Vector() = default;
240
241 explicit Vector(const mxArray *ptr) {
242 n = (int) mxGetNumberOfElements(ptr);
243 vector = (T *) malloc((unsigned) (n * sizeof(T)));
244
245 auto matlab_ptr = mxGetPr(ptr);
246 for (int i = 0; i < n; i++) { vector[i] = (T) matlab_ptr[i]; }
247 }
248
249 bool has_elements() { return vector != nullptr; }
250
251 inline T operator[](int value) const { return vector[value]; };
252
253 inline int size() const { return n; };
254};

311 /**
312 * @brief Return the component of the field corresponding to the index
313 * provided.
314 *
315 * 0 = Ex, 1 = Ey, 2 = Ez, 3 = Hx, 4 = Hy, 5 = Hz.
316 * This is the indexing order that other storage containers use.
317 *
318 * Throws an error if provided an index <0 or >5.
319 *
320 * @param index Field component index to fetch
321 * @return std::complex<double> The field component
322 */
323 std::complex<double> operator[](int index) {
324 switch (index) {
325 case 0:
326 return Ex;
327 break;
328 case 1:
329 return Ey;
330 break;
331 case 2:
332 return Ez;
333 break;
334 case 3:
335 return Hx;
336 break;
337 case 4:
338 return Hy;
339 break;
340 case 5:
341 return Hz;
342 break;
343 default:
344 throw std::runtime_error("Index " + std::to_string(index) +
345 " does not correspond to a field component.");
346 break;
347 }
348 }
