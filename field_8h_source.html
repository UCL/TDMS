16/**
17 * A generic grid entity. For example:
18 *
19 * ______________________ (2, 1, 0)
20 * / / /
21 * / / /
22 * /___________/_________/
23 * (0, 0, 0) (1, 0, 0) (2, 0, 0)
24 *
25 * has I_tot = 2, J_tot = 1, K_tot = 0.
26 *
27 * NOTE: For storage purposes, this means that field values associated to cells
28 * are stored _to the left_. That is, Grid(0,0,0) is associated to the cell
29 * (-1,-1,-1). This is contrary to the way values are associated to cells, where
30 * cell (0,0,0) is associated to the field values (0,0,0).
31 */
32class Grid {
33protected:
34 // the preferred interpolation methods (pim) for interpolating between the
35 // grid values, default is BandLimited
36 PreferredInterpolationMethods pim =
37 PreferredInterpolationMethods::BandLimited;
38
39public:
40 // The {IJK}_tot values of this grid
41 IJKDimensions tot = {0, 0, 0};
42
43 /**
44 * Maximum value out of I_tot, J_tot and K_tot
45 * @return value
46 */
47 int max_IJK_tot() const { return tot.max(); };
48
49 /**
50 * @brief Set the preferred interpolation methods
51 */
52 void set_preferred_interpolation_methods(PreferredInterpolationMethods _pim) {
53 pim = _pim;
54 };
55};
