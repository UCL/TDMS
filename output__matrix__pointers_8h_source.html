57 /**
58 * @brief Throw an error if the matrix_pointer corresponding to any one of the
59 * matrix names passed already points to allocated memory.
60 *
61 * This function exists to avoid memory leaks - it is possible for us to

63 * always validate we have not done this before attempting to assign memory.
64 *
65 * @param matrix_names List of output matrices to check.
66 */
67 void error_on_memory_assigned(const std::vector<std::string> &matrix_names) {
68 if (memory_already_assigned(matrix_names)) {
69 throw std::runtime_error(

71 }
72 }
