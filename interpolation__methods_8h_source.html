58 /**
59  * @brief Defines our order of preference for the use of the various schemes.
60  *
61  * There should never be an instance in which we wish to use BLi to position 7 - this will take us to a point OUTSIDE the computational domain. However for completion purposes (and if we find a use for it), it is included.
62  *
63  * MODIFICATIONS TO THE ALIASED INTS WILL CHANGE THE ORDER OF SCHEME PREFERENCE!
64  *
65  * For band-limited schemes, we have 8 equidistant datapoints and can interpolate to the midpoint of any pair of consecutive points, or "half a point spacing" to the right of the final data point.
66  * These "interpolation positions" (interp positions) are marked with "o" below:
67  * v0 v1 v2 v3 v4 v5 v6 v7
68  * o o o o o o o o
69  * We label these interp positions with the index of the point to the left: a data point at interp position 0 is effectively a data point "v0.5", being at the midpoint of v0 and v1, for example.
70  */
71 enum scheme_value
72 {
73  BAND_LIMITED_0 = 4, // use bandlimited interpolation w/ interp position = 0
74  BAND_LIMITED_1 = 6, // use bandlimited interpolation w/ interp position = 1
75  BAND_LIMITED_2 = 8, // use bandlimited interpolation w/ interp position = 2
76  BAND_LIMITED_3 = 9, // use bandlimited interpolation w/ interp position = 3 [Preferred method if available]
77  BAND_LIMITED_4 = 7, // use bandlimited interpolation w/ interp position = 4
78  BAND_LIMITED_5 = 5, // use bandlimited interpolation w/ interp position = 5
79  BAND_LIMITED_6 = 3, // use bandlimited interpolation w/ interp position = 6
80  BAND_LIMITED_7 = -1, // use bandlimited interpolation w/ interp position = 7 [Only applicable if we want to extend beyond the final Yee cell, current code functionality is to throw an error in the case where this would be used.]
81  BAND_LIMITED_CELL_ZERO = -2, // use bandlimited interpolation to interpolate to the centre of Yee cell 0 [implemented, but current code functionality is to throw an error here]
82  CUBIC_INTERP_MIDDLE = 2, // cubic interpolation to middle 2 of 4 points (interp1)
83  CUBIC_INTERP_FIRST = 1, // cubic interpolation to first 2 of 4 points (interp2)
84  CUBIC_INTERP_LAST = 0 // cubic interpolation to last 2 of 4 points (interp3)
85 };
