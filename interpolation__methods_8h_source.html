14 /**
15  * @brief Defines our order of preference for the use of the various schemes.
16  *
17  * There should never be an instance in which we wish to use BLi to position 7 - this will take us to a point OUTSIDE the computational domain. However for completion purposes (and if we find a use for it), it is included.
18  *
19  * MODIFICATIONS TO THE ALIASED INTS WILL CHANGE THE ORDER OF SCHEME PREFERENCE!
20  *
21  * For band-limited schemes, we have 8 equidistant datapoints and can interpolate to the midpoint of any pair of consecutive points, or "half a point spacing" to the right of the final data point.
22  * These "interpolation positions" (interp positions) are marked with "o" below:
23  * v0 v1 v2 v3 v4 v5 v6 v7
24  * o o o o o o o o
25  * We label these interp positions with the index of the point to the left: a data point at interp position 0 is effectively a data point "v0.5", being at the midpoint of v0 and v1, for example.
26  */
27 enum scheme_value
28 {
29  BAND_LIMITED_0 = 4, // use bandlimited interpolation w/ interp position = 0
30  BAND_LIMITED_1 = 6, // use bandlimited interpolation w/ interp position = 1
31  BAND_LIMITED_2 = 8, // use bandlimited interpolation w/ interp position = 2
32  BAND_LIMITED_3 = 9, // use bandlimited interpolation w/ interp position = 3 [Preferred method if available]
33  BAND_LIMITED_4 = 7, // use bandlimited interpolation w/ interp position = 4
34  BAND_LIMITED_5 = 5, // use bandlimited interpolation w/ interp position = 5
35  BAND_LIMITED_6 = 3, // use bandlimited interpolation w/ interp position = 6
36  BAND_LIMITED_7 = 2, // use bandlimited interpolation w/ interp position = 7
37  BAND_LIMITED_CELL_ZERO = 1, // use bandlimited interpolation to interpolate to the centre of Yee cell 0
38  CUBIC_INTERP_MIDDLE = 0, // cubic interpolation to middle 2 of 4 points (interp1)
39  CUBIC_INTERP_FIRST = -1, // cubic interpolation to first 2 of 4 points (interp2)
40  CUBIC_INTERP_LAST = -2 // cubic interpolation to last 2 of 4 points (interp3)
41 };

84  /**
85  * @brief Executes the interpolation scheme on the data provided
86  *
87  * The interpolation schemes are all of the form
88  * interpolated_value = \sum_{i=0}^{7} scheme_coeffs[i] * v[i],
89  * so provided that the coefficients have been set correctly in construction (and the data gathered appropriately), we can run the same for loop for each interpolation scheme.
90  *
91  * For slight speedup, the actual sum performed loops over those i such that
92  * 0 <= first_nonzero_coeff <= i <= last_nonzero_coeff <= 7.
93  *
94  * @param v Sample datapoints to use in interpolation; v[0] should be the first of 8 values
95  * @param offset [Default 0] Read buffer from v[offset] rather than v[0]
96  * @return double Interpolated value
97  */
98  template<typename T>
99  T interpolate(const T *v, const int offset = 0) const {
100  T interp_value = 0.;
101  for (int ind = first_nonzero_coeff; ind <= last_nonzero_coeff; ind++) {
102  interp_value += scheme_coeffs[ind] * v[ind + offset];
103  }
104  return interp_value;
105  };
106

111  * @return true This scheme has greater value
112  * @return false This scheme has lesser, or equal, value to s
113  */
114  bool is_better_than(const InterpolationScheme &s) const;
