92 /**
93 * @brief Construct a new HDF5Reader for a named file.
94 * @param filename The name of the file.
95 * @throws XX if file is not found.
96 */
97 HDF5Reader(const std::string &filename)
98 : HDF5Base(filename, H5F_ACC_RDONLY) {}
99
100 /**
101 * @brief Reads a named dataset from the HDF5 file.
102 * @param dataname The name of the datset to be read.
103 * @param data A pointer to an array of correct size.
104 */
105 template<typename T>
106 void read(const std::string &dataset_name, T *data) const {
107 spdlog::debug("Reading {} from file: {}", dataset_name, filename_);
108
109 // get the dataset and dataspace (contains dimensionality info)
110 H5::DataSet dataset = file_->openDataSet(dataset_name);
111 H5::DataSpace dataspace = dataset.getSpace();
112 spdlog::debug("Created dataspace");
113
114 // need to get the number of matrix dimensions (rank) so that we can
115 // dynamically allocate `dimensions`
116 int rank = dataspace.getSimpleExtentNdims();
117 spdlog::debug("Rank of dataspace: {}", rank);
118 hsize_t *dimensions = new hsize_t[rank];
119 dataspace.getSimpleExtentDims(dimensions);
120 spdlog::debug("Got dimensions");
121
122 // now get the data type
123 H5::DataType datatype = dataset.getDataType();
124 spdlog::debug("Got datatype");
125 dataset.read(data, datatype);
126 spdlog::debug("Read");
127
128 delete[] dimensions;
129 }
130};
