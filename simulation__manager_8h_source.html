52 /*! Width of the ramp when introducing the waveform in steady state mode */
53 double ramp_width = 4.;
54 /**
55 * @brief Evaluates the linear ramp function r(t)
56 *
57 * r(t) = 1 if t > ramp_width * period
58 * = t / (ramp_width * period) if t <= ramp_width * period
59 *
60 * Period = 1. / (omega_an / (2 * pi)) = 2 * pi / omega_an
61 *
62 * NB Note that r(t) = min(1, t / (ramp_width * period))
63 *
64 * @param t Time to evaluate function at
65 * @return double
66 */
67 double linear_ramp(double t) {
68 double period = 2 * tdms_math_constants::DCPI / inputs.params.omega_an;
69 return std::min(1., t / (ramp_width * period));
70 }
