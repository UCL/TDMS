50 /**
51 * @brief Evaluates the linear ramp function r(t)
52 *
53 * r(t) = 1 if t > ramp_width * period
54 * = t / (ramp_width * period) if t <= ramp_width * period
55 *
56 * Period = 1. / (omega_an / (2 * pi)) = 2 * pi / omega_an
57 *
58 * NB Note that r(t) = min(1, t / (ramp_width * period))
59 *
60 * @param t Time to evaluate function at
61 * @return double
62 */
63 double linear_ramp(double t) {
64 double period = 2 * tdms_math_constants::DCPI / inputs.params.omega_an;
65 return std::min(1., t / (ramp_width * period));
66 }
